 80    @staticmethod
 81    def one_hot_encoding(seq: str):
 82        '''
 83        Return the one hot encoding of a sequence.
 84        Ambiguous nucleotides are encoded as zeros.
 85        Adapted from PROcapNet code.
 86
 87        Keyword arguments:
 88        - seq: Sequence to encode.
 89
 90        Output:
 91        - encoding: numpy array of shape (len(seq), 4)
 92        '''
 93        ambiguous_nucs = ["Y", "R", "W", "S", "K", "M", "D", "V", "H", "B", "X", "N"]
 94
 95        sequence = seq.upper()
 96        if isinstance(sequence, str):
 97            sequence = list(sequence)
 98
 99        alphabet = ["A", "C", "G", "T"]
100        alphabet_lookup = {char: i for i, char in enumerate(alphabet)}
101
102        ohe = np.zeros((len(sequence), len(alphabet)), dtype="int8")
103        for i, char in enumerate(sequence):
104            if char in alphabet:
105                idx = alphabet_lookup[char]
106                ohe[i, idx] = 1
107            else:
108                assert char in ambiguous_nucs, char
109
110        return ohe

127    def get_region_seq(self, chrom: str, start: int, end: int) -> str:
128        '''
129        Return the sequene of a given region. 
130        The coordinates are of bed convention.
131        
132        Keyword arguments:
133        - chrom: Chromosome name.
134        - start: Start coordinate.
135        - end: End coordinate.
136        
137        Returns:
138        - seq: Sequence of the region.
139        '''
140        with open(self.fasta_path, "r") as handle:
141            for record in SeqIO.parse(handle, "fasta"):
142                if record.id == chrom:
143                    return str(record.seq[start:end])  # Convert to 0-based index
144        return None

146    def get_all_region_seqs(self):
147        '''
148        Get the sequences for all regions.
149        This function reads the genome file to memory.
150        
151        Returns:
152        - A list of strings of length region_length
153        '''
154        # get chrom name to seq dict
155        chrom_seq_dict = {}
156        with open(self.fasta_path, "r") as handle:
157            for record in SeqIO.parse(handle, "fasta"):
158                chrom_seq_dict[record.id] = str(record.seq)
159
160        out_seqs = []
161
162        for i, region in enumerate(self.get_region_bed_table().iter_regions()):
163            if not region["chrom"] in chrom_seq_dict:
164                raise ValueError(f"Chromosome {region['chrom']} not found in the genome file.")
165            seq = chrom_seq_dict[region["chrom"]][region["start"]:region["end"]]
166            out_seqs.append(seq)
167
168        out_seq_lens = np.array([len(s) for s in out_seqs])
169        seq_len = out_seq_lens[0]
170        if not (seq_len == out_seq_lens).all():
171            raise ValueError(f"Region sequences have different lengths: {out_seq_lens}")
172        
173        return out_seqs

175    def get_all_region_one_hot(self):
176        '''
177        Get the one hot encoding for all regions.
178        This function reads the genome file to memory 
179        so that it is memory intensive.
180
181        Returns:
182        - An numpy array of size (num_regions, region_length, 4)
183        '''
184        out_seqs = self.get_all_region_seqs()
185        seq_len = len(out_seqs[0])
186
187        out_arr = np.zeros((len(out_seqs), seq_len, 4), dtype="int8")
188
189        for i, seq in enumerate(out_seqs):
190            out_arr[i] = self.one_hot_encoding(seq)
191
192        return out_arr
